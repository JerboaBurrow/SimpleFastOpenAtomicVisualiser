34glm::vec3 cartesian2Spherical(glm::vec3 xyz)
35{
36 float r = glm::length(xyz);
37 float sgn = (-1.0*xyz.z) < 0 ? -1.0 : 1.0;
38 return
39 {
40 r,
41 std::acos(xyz.y/r),
42 sgn * std::acos(xyz.x/std::sqrt(xyz.x*xyz.x+xyz.z*xyz.z))
43 };
44}

32glm::vec3 cartesian2Spherical(glm::vec3 xyz)
33{
34 float r = glm::length(xyz);
35 float sgn = (-1.0*xyz.z) < 0 ? -1.0 : 1.0;
36 return
37 {
38 r,
39 std::acos(xyz.y/r),
40 sgn * std::acos(xyz.x/std::sqrt(xyz.x*xyz.x+xyz.z*xyz.z))
41 };
42}

104std::vector<std::string> split(std::string str, std::regex delim)
105{
106 std::vector<std::string> s;
107 auto matches = std::sregex_iterator(str.begin(), str.end(), delim);
108 if (std::distance(matches, std::sregex_iterator()) == 0) { return {str}; }
109 auto key = matches->position();
110 if (key > 0)
111 {
112 s.push_back(str.substr(0, key-1));
113 }
114 while (matches != std::sregex_iterator())
115 {
116 key = matches->position();
117 matches++;
118 auto next = matches->position();
119 s.push_back(str.substr(key, next-1-key));
120 }
121 return s;
122}

242 "#version " GLSL_VERSION "\n"

244 "in vec2 billboard;\n"
245 "in vec3 aViewPos;\n"
246 "in vec3 bViewPos;\n"

253 "uniform mat4 view;\n"
254 "uniform mat4 proj;\n"
255 "uniform vec4 lightPos;\n"
256 "uniform vec4 lightColour;\n"
257 "uniform float ambientLight;\n"
258 "uniform float bondScale;\n"
259 "bool sphereHit(vec3 rayDirection, vec3 centre, float radius, out vec3 pos, out vec3 normal)\n"
260 "{\n"
261 " float b = 2.0 * dot(rayDirection, -centre);\n"
262 " float r2 = radius*radius;\n"
263 " float determinant = b * b - (4.0 * (dot(centre, centre) - r2));\n"
264 " if(determinant < 0.0) { return false; }\n"
265 " determinant = sqrt(determinant);\n"
266 " pos = rayDirection * min((-b+determinant)*0.5, (-b-determinant)*0.5);\n"
267 " normal = normalize(pos - centre);\n"
268 " return true;\n"
269 "}\n"
270 /*
271 * Creates a capsule (sphere integrated along a line).
272 * .-.B
273 * / / \
274 * / / /
275 * / / /
276 * / / /
277 * / / /
278 * \ / /
279 * A--
280 *
281 * To derive, consider in view space a ray r(t) = t*v, and a capsule with midline c(s) = a+s*n.
282 * Then solve the quadratic ||a-r(t)-[(a-r(t).n)]n||^2-R^2 = 0 (=> r(t)->midline distance = R);
283 * Also then project the hit point to the midline for normals.
284 */
285 "bool capsuleHit"
286 "("
287 " vec3 rayDirection,"
288 " vec3 capsuleDirection,"
289 " vec3 capsuleA,"
290 " vec3 capsuleB,"
291 " float radius,"
292 " float midLength,"
293 " out vec3 viewPos,"
294 " out vec3 viewNormal,"
295 " out vec3 projectedHitPoint,"
296 " out float s"
297 ")\n"
298 "{\n"
299 " float a = dot(capsuleA, capsuleDirection);"
300 " float b = dot(rayDirection, capsuleDirection);\n"
301 " float c = dot(capsuleA, rayDirection);\n"
302 " float l = dot(capsuleA, capsuleA);\n"
303 " float r2 = radius*radius;\n"
304 " float determinant = (a*b-c)*(a*b-c)-(1.0-b*b)*(l-a*a-r2);\n"
305 " if (determinant < 0)\n"
306 " {\n"
307 " if (sphereHit(rayDirection, capsuleA, radius, viewPos, viewNormal)){ projectedHitPoint = capsuleA; s = 0.0; return true; }\n"
308 " if (sphereHit(rayDirection, capsuleB, radius, viewPos, viewNormal)){ projectedHitPoint = capsuleB; s = midLength; return true; }\n"
309 " return false;"
310 " \n}"
311 " float d = sqrt(determinant);\n"
312 // Ray hit point.
313 " float t = min( (-(a*b-c)+d)/(1.0-b*b), (-(a*b-c)-d)/(1.0-b*b) );"
314 " viewPos = rayDirection * t;\n"
315 " s = -dot(capsuleA-viewPos, capsuleDirection);\n"
316 " if (s < 0){\n"
317 " if (sphereHit(rayDirection, capsuleA, radius, viewPos, viewNormal)){ projectedHitPoint = capsuleA; s = 0.0; return true; }\n"
318 " return false;"
319 " \n}"
320 " if (s > midLength){\n"
321 " if (sphereHit(rayDirection, capsuleB, radius, viewPos, viewNormal)){ projectedHitPoint = capsuleB; s = midLength; return true; }\n"
322 " return false;"
323 " }"
324 " vec3 u = s*capsuleDirection;\n"
325 " projectedHitPoint = capsuleA+u;\n"
326 " viewNormal = normalize(viewPos-projectedHitPoint);\n"
327 " return true;\n"
328 "}"
329 "void main()\n"
330 "{\n"
331 " vec3 lightViewPos = (view*lightPos).xyz;\n"
332 " vec3 rayDirection = normalize(vec3(billboard * bondScale, 0.0) + comViewPos);"
333 " float midLength = length(bViewPos-aViewPos);\n"
334 " vec3 capsuleDirection = normalize(bViewPos-aViewPos);\n"
335 " vec3 viewPos; vec3 viewNormal; vec3 projectedHitPoint; float s;\n"
336 " bool hit = capsuleHit(rayDirection, capsuleDirection, aViewPos, bViewPos, bondScale, midLength, viewPos, viewNormal, projectedHitPoint, s);\n"
337 " if (!hit) { discard; }\n"
338 " else\n"
339 " {\n"
340 " vec4 clipPos = proj * vec4(viewPos, 1.0);\n"
341 " float ndcDepth = clipPos.z / clipPos.w;\n"
342 " gl_FragDepth = ((gl_DepthRange.diff * ndcDepth) + gl_DepthRange.near + gl_DepthRange.far) / 2.0;\n"
343 " float diff = max(dot(viewNormal, normalize(lightViewPos-projectedHitPoint)), 0.0);\n"
344 " vec4 col = a_colour;\n"
345 " if (s > 0.5*midLength) { col = b_colour; }\n"
346 " colour = vec4((ambientLight + diff)*lightColour.rgb * col.rgb, col.a);\n"
347 " }\n"

204 void updateAtoms(const std::vector<Atom> & atoms, std::map<uint64_t, uint8_t> levelsOfDetail = {})
205 {
206 uint64_t i = 0;
207 uint64_t furthest = 0;
208 float maxDistance = 0.0f;
209 float minDistance = 1e9;
210
211 for (const Atom & atom : atoms)
212 {
213 // scalar projection of atom position onto camera vector
214 cameraDistances[i] = glm::dot(atom.position-cameraPosition, glm::normalize(-cameraPosition));
215 if (cameraDistances[i] > maxDistance) { furthest = i; }
216 maxDistance = std::max(maxDistance, cameraDistances[i]);
217 minDistance = std::min(minDistance, cameraDistances[i]);
218 i++;
219 }
220
221 for (auto & buf : buffers) { buf->flip(); }
222
223 float rcrit = 0.5f*(maxDistance-minDistance);
224 for (uint64_t i = 0; i < atoms.size(); i++)
225 {
226 auto lod = levelsOfDetail.find(i);
227 if (lod == levelsOfDetail.end())
228 {
229 float r = glm::length(atoms[furthest].position-atoms[i].position);
230 if (r < rcrit)
231 {
232 buffers[0]->insert(atoms[i]);
233 }
234 else
235 {
236 buffers[levelOfDetail]->insert(atoms[i]);
237 }
238 }
239 else
240 {
241 uint8_t l = lod->second;
242 if (l > meshes.size()-1) { l = meshes.size()-1; }
243 buffers[l]->insert(atoms[i]);
244 }
245 }
246
247 for (auto & buf : buffers) { buf->updateVertexArray(); }
248 }

316template <class T>
317void centreMesh(std::vector<Trixel<T>> & mesh)
318{
319 vec3<T> com = vec3<T>(0.0);
320 for (const auto & tri : mesh)
321 {
322 com += tri.x + tri.y + tri.z;
323 }
324 com /= float(mesh.size()*3);
325 for (auto & tri : mesh){
326 tri.x -= com;
327 tri.y -= com;
328 tri.z -= com;
329 }
330}
